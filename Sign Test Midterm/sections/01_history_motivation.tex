            \textbf{John Arbuthnot (1667-1735)}
            \begin{itemize}\itemsep0pt
                \item Bác sĩ Hoàng gia Anh
                \item Nhà toán học, nhà văn
                \item Nghiên cứu: Tỷ lệ sinh nam/nữ ở London
            \end{itemize}
            
            \vspace{0.15cm}
            \textbf{Phát hiện đáng chú ý:}
            \begin{itemize}\itemsep0pt
                \item \alert{82 năm liên tiếp} (1629-1710)
                \item Số trẻ em nam sinh > Số trẻ em nữ
                \item Ngẫu nhiên hay có ý nghĩa?
            \end{itemize}
            
            \vspace{0.15cm}
            \textbf{Kết luận lịch sử:}
            \begin{itemize}\itemsep0pt
                \item Nếu ngẫu nhiên: $P = (1/2)^{82} \approx 10^{-25}$
                \item → \alert{Bác bỏ giả thuyết ngẫu nhiên!}
                \item Một trong những kiểm định đầu tiên
            \end{itemize}
        \end{column}
