\subsection{Tóm tắt}

\begin{frame}[allowframebreaks,shrink=5]{Take-home Messages}
    \footnotesize
    
    \begin{block}{1. Sign Test = Simple but Powerful}
        \begin{itemize}\itemsep0pt
            \item 300+ năm lịch sử, vẫn cực kỳ hữu ích
            \item Dễ hiểu, dễ thực hiện
            \item Không cần giả định phức tạp
        \end{itemize}
    \end{block}
    
    \vspace{0.2cm}
    
    \begin{block}{2. Khi nào nên dùng Sign Test?}
        \begin{itemize}\itemsep0pt
            \item \checkmark Dữ liệu \textbf{không phân phối chuẩn}
            \item \checkmark Có \textbf{outliers} hoặc extreme values
            \item \checkmark Quan tâm đến \textbf{median} hơn mean
            \item \checkmark Mẫu nhỏ (n = 10-30)
            \item \checkmark Cần phương pháp \textbf{robust}
        \end{itemize}
    \end{block}
    
    \vspace{0.2cm}
    
    \begin{block}{3. Python makes it easy!}
        \begin{itemize}\itemsep0pt
            \item Chỉ cần \texttt{scipy.stats.binomtest}
            \item Hiểu logic → Code 5 dòng
            \item Demo trong Jupyter Notebook rất trực quan
        \end{itemize}
    \end{block}
\end{frame}
